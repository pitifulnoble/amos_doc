# java实现生产消费

## producer
```java
package com.amos.simple2.kafka;

import org.apache.kafka.clients.producer.KafkaProducer;
import org.apache.kafka.clients.producer.ProducerRecord;
import org.apache.kafka.common.serialization.StringSerializer;

import java.util.Properties;

/**
 * @program: simple2
 * @description:
 * java 实现生产者
 * @author: wangyuqing3
 * @created: 2021/03/22 00:46
 */
public class ProducerDemo {
    public static void main(String[] args) {
        final Properties prop = new Properties();
        // kafka地址
        prop.put("bootstrap.servers","10.211.55.8:9092");
        // 指定key-value数据序列化格式
        prop.put("key.serializer",StringSerializer.class.getName());
        prop.put("value.serializer",StringSerializer.class.getName());
        // kafka生产者
        final KafkaProducer<String, String> producer = new KafkaProducer<String, String>(prop);
        // 向topic中产生数据
        String topic = "hello";
        producer.send(new ProducerRecord<>(topic,"hello kafka"));
        // 关闭kafka连接
        producer.close();
    }
}
```

## consumer
```java
package com.amos.simple2.kafka;

import org.apache.kafka.clients.consumer.ConsumerRecord;
import org.apache.kafka.clients.consumer.ConsumerRecords;
import org.apache.kafka.clients.consumer.KafkaConsumer;
import org.apache.kafka.common.serialization.StringDeserializer;

import java.time.Duration;
import java.util.ArrayList;
import java.util.List;
import java.util.Properties;

/**
 * @program: simple2
 * @description:
 * @author: wangyuqing3
 * @created: 2021/03/22 00:54
 */
public class ConsumerDemo {
    public static void main(String[] args) {
        final Properties prop = new Properties();
        // kafka地址
        prop.put("bootstrap.servers", "10.211.55.8:9092");
        // 指定反序列化类型
        prop.put("key.deserializer", StringDeserializer.class.getName());
        prop.put("value.deserializer", StringDeserializer.class.getName());
        // 指定消费者组
        prop.put("group.id", "con-1");
        // 创建消费者
        final KafkaConsumer<String, String> consumer = new KafkaConsumer<>(prop);

        List<String> topics = new ArrayList<>();
        topics.add("hello");
        // 订阅指定的topic
        consumer.subscribe(topics);
        while (true){
            // 消费数据
            ConsumerRecords<String, String> poll = consumer.poll(Duration.ofSeconds(1));
            for(ConsumerRecord<String, String> consumerRecord: poll){
                System.out.println(consumerRecord);
            }
        }
    }
}
```
