## 01.other_bucket_key
用在filter过滤器中，用来指定 剩余未过滤的文档

```json
{
  "aggs": {
    "3": {
      "date_histogram": {
        "field": "@timestamp",
        "interval": "1d",
        "time_zone": "Asia/Shanghai",
        "min_doc_count": 1
      },
      "aggs": {
        "4": {
          "filters": {
            "filters": {
              "geo.src.keyword:CN": {
                "query_string": {
                  "query": "geo.src.keyword:CN",
                  "analyze_wildcard": true,
                  "default_field": "*"
                }
              },
              "geo.src.keyword:IN": {
                "query_string": {
                  "query": "geo.src.keyword:IN",
                  "analyze_wildcard": true,
                  "default_field": "*"
                }
              },
              "others": {
                "query_string": {
                  "query": "others",
                  "analyze_wildcard": true,
                  "default_field": "*"
                }
              }
            },
            "other_bucket_key": "others"
          },
          "aggs": {
            "2": {
              "avg": {
                "field": "bytes"
              }
            }
          }
        }
      }
    }
  }
}
```
